import json
from unittest.mock import AsyncMock, patch

import pytest


@pytest.mark.asyncio
async def test_reflexion_recovers_from_bad_output_after_retry():
    bad = {"choices": [{"message": {"tool_calls": [{"id": "x", "function": {"name": "reflexion_result", "arguments": "{}"}}]}}]}
    valid = {
        "scores": {"clarity": 5, "depth": 5, "specificity": 5, "STAR_completeness": 5},
        "strengths": [], "gaps": [], "improvement_suggestion": "x",
        "follow_up_question": "y", "weakness_tags_to_add": [],
    }
    good = {"choices": [{"message": {"tool_calls": [{"id": "x", "function": {"name": "reflexion_result", "arguments": json.dumps(valid)}}]}}]}
    counter = {"n": 0}
    async def fake(*a, **kw):
        counter["n"] += 1
        return bad if counter["n"] <= 1 else good
    with patch("app.services.reflexion.evaluator.chat_complete", new=fake):
        from app.services.reflexion.evaluator import run_reflexion
        r = await run_reflexion("q", "a")
    assert r.scores["clarity"] == 5


@pytest.mark.asyncio
async def test_reflexion_handles_low_scores_writes_weakness():
    payload = {
        "scores": {"clarity": 3, "depth": 4, "specificity": 4, "STAR_completeness": 3},
        "strengths": [], "gaps": ["全面不足"],
        "improvement_suggestion": "全部改善",
        "follow_up_question": "再来一题",
        "weakness_tags_to_add": [
            {"tag": "tech-基础", "category": "tech", "severity": 0.8},
            {"tag": "star-不完整", "category": "star", "severity": 0.7},
        ],
    }
    resp = {"choices": [{"message": {"tool_calls": [{"id": "x", "function": {"name": "reflexion_result", "arguments": json.dumps(payload)}}]}}]}
    with patch("app.services.reflexion.evaluator.chat_complete", new=AsyncMock(return_value=resp)):
        from app.services.reflexion.evaluator import run_reflexion
        r = await run_reflexion("q", "a")
    assert len(r.weakness_tags_to_add) == 2
    assert all(0 <= w.severity <= 1 for w in r.weakness_tags_to_add)

## 分级记忆设计

| 层 | 存储 | 写时机 | 读时机 |
|---|---|---|---|
| L0 短期 | LangGraph state.messages | 每条消息 | 当前对话 |
| L1 工作记忆 | `interview_session.summary` | Coach 后 Celery 异步 | 下场开场 |
| L2 长期画像 | `user_profile` + pgvector | Reflexion 后 EMA | 每个 Agent 节点开场 |
| L3 STAR 故事库 | `star_stories` + pgvector | Reflexion 抽 STAR | 出题前检索 |
| L4 弱点标签 | `weakness_tags` | Reflexion 低分时 | 出题加权 |

## Reflexion 链路

```
[Tech Agent 出题] → [用户回答] → [Reflexion]
                                    │
                                    ├── scores（clarity/depth/specificity/STAR）
                                    ├── strengths / gaps
                                    ├── improvement_suggestion
                                    ├── follow_up_question
                                    └── weakness_tags_to_add → L4
                                                              ↓
                              L2 EMA ← averages ← session.scores
                                                              ↓
                                          interview_message.reflexion_payload
                                                              ↓
                                                star_extracted? → L3 STAR
```

## 📚 实现说明（招聘官关注的工程亮点）

1. **LangGraph StateGraph 顺序编排** — `app/services/agents/graph.py:1` 5 节点顺序 + `astream_events` 接 SSE。
2. **MemGPT 分级记忆** — `app/services/memory/` L0 state、L2 EMA 增量更新（α=0.3）、L3 pgvector 去重（cosine_distance < 0.15）、L4 occurrence_count 累加。
3. **Reflexion 单轮闭环** — `app/services/reflexion/` Pydantic + OpenAI Function Call 双重约束 + 3 次重试。
4. **跨会话记忆生效** — Celery `summarize_session` 写 L1 摘要 → 下场 HR Agent 开场注入。
5. **RAG 工程** — pgvector cosine_distance + chunk overlap 100 token + Firecrawl 抓 5 个权威文档源。
