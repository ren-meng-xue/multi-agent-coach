- [ ] **Step 2: 写 `frontend/lib/hooks/useInterview.ts`**

```typescript
"use client";

import { useAuth } from "@clerk/nextjs";
import { useCallback, useRef, useState } from "react";
import { readSse } from "@/lib/sse";
import type { AgentRole, Message } from "@/lib/types";

export function useInterview(sessionId: string, firstQuestion: string) {
  const { getToken } = useAuth();
  const [messages, setMessages] = useState<Message[]>(
    firstQuestion ? [{ role: "hr_agent", content: firstQuestion }] : []
  );
  const [streaming, setStreaming] = useState(false);
  const [phase, setPhase] = useState<string>("hr");
  const [done, setDone] = useState(false);
  const abortRef = useRef<AbortController | null>(null);

  const send = useCallback(
    async (userAnswer: string) => {
      if (!userAnswer.trim() || streaming) return;
      const userMsg: Message = { role: "user", content: userAnswer };
      setMessages((p) => [...p, userMsg]);
      setStreaming(true);

      abortRef.current?.abort();
      const controller = new AbortController();
      abortRef.current = controller;

      try {
        const token = await getToken();
        for await (const ev of readSse(`/api/backend/v1/interview/${sessionId}/answer`, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Accept: "text/event-stream",
            Authorization: `Bearer ${token}`,
          },
          body: JSON.stringify({ user_answer: userAnswer }),
          signal: controller.signal,
        })) {
          if (ev.event === "agent_message") {
            const role = (ev.data.role || "hr_agent") as AgentRole;
            setMessages((p) => [...p, { role, content: ev.data.content, phase: ev.data.phase }]);
            if (ev.data.phase) setPhase(ev.data.phase);
          } else if (ev.event === "done") {
            setDone(true);
            setPhase("done");
          }
        }
      } finally {
        setStreaming(false);
      }
    },
    [sessionId, streaming]
  );

  const cancel = useCallback(() => {
    abortRef.current?.abort();
    setStreaming(false);
  }, []);

  return { messages, streaming, phase, done, send, cancel };
}
```

"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";

export function InputArea({
  onSend,
  disabled,
}: {
  onSend: (text: string) => void;
  disabled: boolean;
}) {
  const [text, setText] = useState("");
  function submit() {
    if (!text.trim() || disabled) return;
    onSend(text);
    setText("");
  }
  return (
    <div className="sticky bottom-0 bg-background border-t px-6 py-3 flex gap-2">
      <Textarea
        value={text}
        onChange={(e) => setText(e.target.value)}
        onKeyDown={(e) => {
          if ((e.metaKey || e.ctrlKey) && e.key === "Enter") submit();
        }}
        placeholder="输入回答 (⌘+Enter 发送)"
        rows={3}
        disabled={disabled}
      />
      <Button onClick={submit} disabled={disabled || !text.trim()} size="lg">
        发送
      </Button>
    </div>
  );
}

function InterviewRoom() {
  const { sessionId } = useParams() as { sessionId: string };
  const sp = useSearchParams();
  const firstQ = sp.get("q") || "";
  const router = useRouter();
  const { messages, streaming, phase, done, send } = useInterview(sessionId, firstQ);
  const bottomRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });

  return (
    <div className="min-h-screen flex flex-col">
      <TopBar phase={phase} />
      <main className="flex-1 overflow-auto">

        <div ref={bottomRef} />

            查看复盘报告 →
          </Button>
        </div>
      ) : (
        <InputArea onSend={send} disabled={streaming} />

  );
}

"use client";

import { useEffect, useState } from "react";
import { Button } from "@/components/ui/button";
import { Sheet, SheetContent, SheetHeader, SheetTitle, SheetTrigger } from "@/components/ui/sheet";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from "@/components/ui/collapsible";
import { Badge } from "@/components/ui/badge";
import type { Profile, StarStory, WeaknessTag } from "@/lib/types";

export function ReviewDrawer({ userId, sessionId }: { userId: number; sessionId: string }) {
  const [open, setOpen] = useState(false);
  const [profile, setProfile] = useState<Profile | null>(null);
  const [stars, setStars] = useState<StarStory[]>([]);
  const [weaknesses, setWeaknesses] = useState<WeaknessTag[]>([]);
  const [loading, setLoading] = useState(false);

  async function load() {
    setLoading(true);
    try {
      const [p, s, w] = await Promise.all([
        fetch(`/api/backend/v1/profile/me?user_id=${userId}`).then((r) => (r.ok ? r.json() : null)),
        fetch(`/api/backend/v1/stars/me?user_id=${userId}`).then((r) => (r.ok ? r.json() : [])),

      setWeaknesses(w);
    } finally {
      setLoading(false);
    }
  }

  useEffect(() => {
    if (open) load();
  }, [open]);

  return (
    <Sheet open={open} onOpenChange={setOpen}>
      <SheetTrigger asChild>
        <Button

          className="fixed bottom-24 right-6 rounded-full shadow-lg z-20"

```python
async def update_session_scores(db: AsyncSession, session_id: uuid.UUID, items: list[dict]) -> None:
    if not items:
        return
    dims = ["clarity", "depth", "specificity", "STAR_completeness"]
    averages: dict[str, float] = {}
    for d in dims:
        vals = [it.get("scores", {}).get(d) for it in items if it.get("scores", {}).get(d) is not None]
        if vals:
            averages[d] = sum(vals) / len(vals)
    s = await get_session(db, session_id)
    if s:
        s.scores = averages
        await db.flush()
```
