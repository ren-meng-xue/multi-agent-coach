def ema_update_scalar(prev: float | None, new: float, alpha: float = 0.3) -> float:
    if prev is None:
        return float(new)
    return float(alpha * new + (1 - alpha) * prev)


def ema_update_dict(
    prev: dict[str, float] | None,
    new: dict[str, float],
    alpha: float = 0.3,
) -> dict[str, float]:
    prev = prev or {}
    out = dict(prev)
    for k, v in new.items():
        out[k] = ema_update_scalar(prev.get(k), v, alpha=alpha)
    return out
```

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.core import UserProfile
from app.services.memory.ema import ema_update_dict, ema_update_scalar


async def get_profile(db: AsyncSession, user_id: int) -> UserProfile | None:
    r = await db.execute(select(UserProfile).where(UserProfile.user_id == user_id))
    return r.scalar_one_or_none()


async def upsert_profile(db: AsyncSession, user_id: int, **fields) -> UserProfile:
    p = await get_profile(db, user_id)
    if p is None:
        p = UserProfile(user_id=user_id, **fields)
        db.add(p)
    else:
        for k, v in fields.items():
            if v is not None:
                setattr(p, k, v)
    await db.flush()
    return p


async def apply_ema_update(
    db: AsyncSession,
    user_id: int,
    strengths_delta: dict[str, float] | None = None,
    weaknesses_delta: dict[str, float] | None = None,
    star_completeness_delta: float | None = None,
    alpha: float = 0.3,
) -> UserProfile:
    p = await get_profile(db, user_id)
    if p is None:
        p = UserProfile(user_id=user_id)
        db.add(p)
        await db.flush()
    if strengths_delta:
        p.tech_strengths = ema_update_dict(p.tech_strengths or {}, strengths_delta, alpha=alpha)
    if weaknesses_delta:
        p.tech_weaknesses = ema_update_dict(p.tech_weaknesses or {}, weaknesses_delta, alpha=alpha)
    if star_completeness_delta is not None:
        p.star_completeness_score = ema_update_scalar(
            p.star_completeness_score, star_completeness_delta, alpha=alpha
        )
    await db.flush()
    return p


async def append_history_entry(
    db: AsyncSession, user_id: int, entry: dict, max_keep: int = 10
) -> None:
    p = await get_profile(db, user_id)
    if p is None:
        p = UserProfile(user_id=user_id, history=[entry])
        db.add(p)
    else:
        hist = list(p.history or [])
        hist.append(entry)
        if len(hist) > max_keep:
            hist = hist[-max_keep:]
        p.history = hist
    await db.flush()


async def increment_total_interviews(db: AsyncSession, user_id: int) -> None:
    p = await get_profile(db, user_id)
    if p is None:
        p = UserProfile(user_id=user_id, total_interviews=1)
        db.add(p)
    else:
        p.total_interviews = (p.total_interviews or 0) + 1
    await db.flush()
```

from sqlalchemy import desc, select
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.core import InterviewSession, UserProfile, WeaknessTag
from app.services.memory.profile_repo import get_profile


async def load_memory(
    db: AsyncSession, user_id: int, topic: str | None = None
) -> dict:
    profile = await get_profile(db, user_id)
    profile_block = None
    if profile:
        profile_block = {
            "goals": profile.goals,
            "experience_summary": profile.experience_summary,
            "tech_strengths": profile.tech_strengths or {},
            "tech_weaknesses": profile.tech_weaknesses or {},
            "star_completeness_score": profile.star_completeness_score,
            "total_interviews": profile.total_interviews,
        }

    # active weaknesses
    w_q = await db.execute(
        select(WeaknessTag)
        .where(WeaknessTag.user_id == user_id, WeaknessTag.status == "active", WeaknessTag.severity > 0.5)
        .order_by(desc(WeaknessTag.severity))
        .limit(10)
    )
    weaknesses = [
        {
            "tag": w.tag,
            "category": w.category,
            "severity": w.severity,
            "occurrence_count": w.occurrence_count,
        }
        for w in w_q.scalars().all()
    ]

    # last session summary
    s_q = await db.execute(
        select(InterviewSession)
        .where(InterviewSession.user_id == user_id, InterviewSession.ended_at.is_not(None))
        .order_by(desc(InterviewSession.ended_at))
        .limit(1)
    )
    last = s_q.scalar_one_or_none()

    # relevant_stars (L3) — D5 才填充，D4 先返回 history 里最近一项
    relevant_stars: list[dict] = []
    if profile and profile.history:
        relevant_stars = list(profile.history)[-3:]

    return {
        "profile": profile_block,
        "relevant_stars": relevant_stars,
        "active_weaknesses": weaknesses,
        "last_session_summary": last.summary if last else None,
    }
```

configure_logging(level="INFO")
log = get_logger("app.workers.profile_tasks")


@celery_app.task(name="app.workers.profile_tasks.summarize_session")
def summarize_session(session_id_str: str) -> str:
    asyncio.run(_summarize(session_id_str))
    return "ok"


async def _summarize(session_id_str: str) -> None:
    session_id = uuid.UUID(session_id_str)
    async with async_session_factory() as db:
        s = await get_session(db, session_id)
        if not s:
            log.warning("summarize_no_session", session=session_id_str)
            return
        msgs = await list_messages(db, session_id)
        text = "\n".join(f"{m.role}: {m.content}" for m in msgs)

        resp = await chat_complete(
            [
                {"role": "system", "content": "请用 200 字以内总结这场面试的关键信息：候选人讲到的项目、技术栈、表达问题。"},
                {"role": "user", "content": text[:8000]},
            ],
            temperature=0.3,
        )
        summary = resp["choices"][0]["message"]["content"]

        s.summary = summary
        await db.flush()
        await append_history_entry(
            db,
            user_id=s.user_id,
            entry={"session_id": str(session_id), "summary": summary},
        )
        await increment_total_interviews(db, user_id=s.user_id)
        await db.commit()
        log.info("summarize_done", session=session_id_str)
```
